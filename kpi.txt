*tạo 1 container chứa postgresql và 1 db tên kong
docker run -d --name kong-database --network=kong-net -p 5555:5432 -e "POSTGRES_USER=kong" -e "POSTGRES_DB=kong" -e "POSTGRES_PASSWORD=kong" postgres:latest

*migrations bootstrap cho db kong không bị rỗng
docker run --rm --network=kong-net -e "KONG_DATABASE=postgres" -e "KONG_PG_HOST=kong-database" -e "KONG_PG_USER=kong" -e "KONG_PG_PASSWORD=kong" kong:latest kong migrations bootstrap

*tạo 1 container cho kong api gateway database là kong(nếu database rỗng thì container sẽ không chạy)
docker run -d --name kong --network=kong-net --link kong-database:kong-database -e "KONG_DATABASE=postgres" -e "KONG_PG_HOST=kong-database" -e "KONG_PG_PASSWORD=kong" -e "KONG_PROXY_ACCESS_LOG=/dev/stdout" -e "KONG_ADMIN_ACCESS_LOG=/dev/stdout" -e "KONG_PROXY_ERROR_LOG=/dev/stderr" -e "KONG_ADMIN_ERROR_LOG=/dev/stderr" -e "KONG_ADMIN_LISTEN=0.0.0.0:8001, 0.0.0.0:8444 ssl" -p 8000:8000 -p 8443:8443 -p 8001:8001 -p 8444:8444 kong:latest

*truy cập vào db kong trong container chứa postgresql
docker exec -it 7305fef5d6b0 psql -U kong -d kong

docker build -f projectfolder\Dockerfile --force-rm -t name .

docker run -d --name api2 --network=kong-net
